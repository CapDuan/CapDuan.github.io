const postRequest = {
  url: pm.variables.get("localhost") + "/apis/v1/system/login/",
  method: 'POST',
  header: {
    'Content-Type': 'application/json',
    'X-Foo': 'bar'
  },
  body: {
    mode: 'raw',
    raw: {
      username: "duanwenchao@medlinker.com",
      password: "d4985765"
    }
  }
}
pm.sendRequest(postRequest, function (err, response) {
  let jwtToken = 'JWT ' + response.json().data.token
  pm.globals.set("token", jwtToken)
});

实现领域驱动设计

DDD领域驱动设计
核心域

第一章入门

什么是领域模型？

是一种==软件模型==，通过==对象模型==来实现，对象包含数据和行为以及关系，并可以描述准确的业务关系

不同的角色：初中高开发，领域专家，项目经理
项目开发就是翻译，客观业务抽象形成产品语言进而翻译为机器语言

为什么需要DDD？

  1. 准确表达业务规则
  2. 开发人员更替时依托模型成本底低
  3. 降低沟通成本，领域专家，开发者，软件。统一描述语言
  4. 战略设计理解那些投入最重要，区分优先级
  5. 战术设计创建DDD部件

业务价值：

真正具有业务价值的软件，符合业务战略，并可以将竞争优势融合在解决方案中，软件不关乎技术，关乎业务。

领域对象和数据对象和贫血对象
通用语言
界限上下文
数据持有器：data holder
逻辑自洽，流程闭环，数据完整

DDD的业务价值：

  1. 领域模型
  2. 业务有更准确的定义和理解
  3. 软件设计做出贡献
  4. 用户体验
  5. 模型边界
  6. 企业架构
  7. 敏捷迭代持续建模
  8. 战略战术新工具

战术建模工具：

  1. 聚合
  2. 实体
  3. 值对象
  4. 领域服务
  5. 领域事件
